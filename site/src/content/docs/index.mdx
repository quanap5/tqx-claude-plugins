import { CardGrid, LinkCard } from '@astrojs/starlight/components';
import catalog from '../../data-catalog.json';

export const groups = [
  { type: 'skill', label: 'Skills', dir: 'skills' },
  { type: 'command', label: 'Commands', dir: 'commands' },
  { type: 'agent', label: 'Agents', dir: 'agents' },
];

## Quick install

```text
/plugin marketplace add quanap5/tqx-claude-plugins
/plugin install tqx@tqx-tools
```

After install, skills auto-trigger on intent; commands and agents are callable as `tqx:<name>`.

## Catalog

{groups.map((g) => {
  const items = catalog.filter((c) => c.type === g.type);
  if (!items.length) return null;
  return (
    <>
      <h3 id={g.dir}>{g.label} <span style="color:var(--sl-color-gray-3);font-weight:400">({items.length})</span></h3>
      <CardGrid>
        {items.map((c) => (
          <LinkCard
            title={c.fqid + (c.vendored ? '  ·  vendored' : '')}
            href={`/tqx-claude-plugins/${c.plural}/${c.slug}/`}
            description={(c.vendored ? `↗ from ${c.source} — ` : '') + c.short}
          />
        ))}
      </CardGrid>
    </>
  );
})}
